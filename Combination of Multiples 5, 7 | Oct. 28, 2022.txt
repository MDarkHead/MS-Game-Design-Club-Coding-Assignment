x=1

seven=7
multiplier=1

while x<=10000:
	five*=multiplier
	seven*=multiplier
	y= five+seven
	print(" " +str(five) + "+" + str(seven) + ": ")
	print("Sum is: " + str(y))
	multiplier += 1
	five=5
	seven=7
	x+=1
